# EOLocale.Text

## Basic

```jsx
import { EOLocale } from 'eo-locale'

<EOLocale.Provider language="en" locales={[]}>
  <EOLocale.Text id="hello" defaultMessage="Hello {name}!" name="World" />
</EOLocale.Provider>
```

## Plural

```jsx
import { EOLocale } from 'eo-locale'

<EOLocale.Provider language="en" locales={[]}>
  <EOLocale.Text
    count="4"
    defaultMessage="{count, plural, one {You have one item.} other {You have {count} items.}}"
    id="hello"
  />
</EOLocale.Provider>
```
