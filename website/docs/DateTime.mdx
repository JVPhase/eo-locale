`DateTime` component provides ability to format values of `Date`.
Full list of supported properties you can find on [MDN](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/DateTimeFormat) page

```jsx
import { DateTime } from '@eo-locale/react';

export function SomeComponent() {
  return (
    <DateTime
      day="numeric"
      language="en"
      month="long"
      value={new Date()}
      weekday="long"
      year="numeric"
    />
  );
}
```
