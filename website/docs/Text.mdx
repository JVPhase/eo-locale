`EOLocale.Text` component provides ability to format string values.
It supports ICU Message Syntax. More information you can find on [official website](http://userguide.icu-project.org/).

```jsx
import { EOLocale } from 'eo-locale'

export function SomeComponent() {
  return (
    <EOLocale.Provider language="en" locales={[]}>
      <EOLocale.Text id="hello" defaultMessage="Hello {name}!" name="World" />
    </EOLocale.Provider>
  );
}
```
